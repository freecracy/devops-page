# 本地隔离环境

# 镜像

FROM centos

RUN dnf -y --disablerepo '*' --enablerepo=extras swap centos-linux-repos centos-stream-repos && dnf -y distro-sync && dnf -y update && dnf -y install openssh-server git && echo "root:$PASSWORD" | chpasswd && ssh-keygen -t ed25519 -q -N '' -f /etc/ssh/ssh_host_ed25519_key

EXPOSE 22

CMD ["/usr/sbin/sshd", "-D"]

# 配置

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: sshd
  labels:
    app: sshd
spec:
  replicas: 1
  selector:
    matchLabels:
      app: sshd
  template:
    metadata:
      labels:
        app: sshd
    spec:
      volumes:
        - name: data
          hostPath:
            path: /Users/tony/local/data/cn
            type: "DirectoryOrCreate"
      containers:
        - name: sshd
          image: sshd
          imagePullPolicy: IfNotPresent
          ports:
            - containerPort: 22
          volumeMounts:
            - name: data
              mountPath: /root
---
apiVersion: v1
kind: Service
metadata:
  name: sshd
spec:
  type: NodePort
  ports:
    - port: 22
      protocol: TCP
      targetPort: 22
      nodePort: 30022
  selector:
    app: sshd
